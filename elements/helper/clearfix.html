Helper classes - clearfix

Clear the float on any element. 

  <div class="clearfix">...</div>
  

// Mixin itself
.clearfix() {
  &:before,
  &:after {
    content: " ";
    display: table;
  }
  &:after {
    clear: both;
  }
}

// Usage as a Mixin
.element {
  .clearfix();
}
